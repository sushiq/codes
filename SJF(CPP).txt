#include <iostream>
#include <climits>
using namespace std;

int main() {
    int n;
    cout << "Enter number of processes: ";
    cin >> n;

    int pid[n], at[n], bt[n], wt[n], tat[n], ct[n];
    bool done[n];  

   
    for (int i = 0; i < n; i++) {
        cout << "Enter Arrival Time and Burst Time for Process P" << i + 1 << " (Example: 0 5): ";
        pid[i] = i + 1;
        cin >> at[i] >> bt[i];
        done[i] = false;  
    }

    int time = 0, completed = 0;
    float totalWT = 0, totalTAT = 0;

    // SJF Scheduling Logic
    while (completed < n) {
        int shortest = -1;
        int minBurst = INT_MAX;

       
        for (int i = 0; i < n; i++) {
            if (!done[i] && at[i] <= time && bt[i] < minBurst) {
                minBurst = bt[i];
                shortest = i;
            }
        }

        if (shortest == -1) {  
            time++;
            continue;
        }

        
        time += bt[shortest];
        ct[shortest] = time;
        tat[shortest] = ct[shortest] - at[shortest];  
        wt[shortest] = tat[shortest] - bt[shortest];  

        totalWT += wt[shortest];
        totalTAT += tat[shortest];

        done[shortest] = true;
        completed++;
    }

   
    cout << "\nProcess\tAT\tBT\tWT\tTAT\tCT\n";
    for (int i = 0; i < n; i++) {
        cout << "P" << pid[i] << "\t" << at[i] << "\t" << bt[i] << "\t" 
             << wt[i] << "\t" << tat[i] << "\t" << ct[i] << endl;
    }

    cout << "\nAverage Waiting Time: " << totalWT / n << endl;
    cout << "Average Turnaround Time: " << totalTAT / n << endl;

    return 0;
}
